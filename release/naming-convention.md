Naming convention
=================

xui [component][_][element] -[modifier][_][element]

component:

	button
	app-bar
		
component with element:

	app-bar_brand
	app-bar_content
	app-bar > brand
	app-bar > content
	app-bar > _brand
	app-bar > _content

modifier:
	
	-left
	-effect-ripple
	-toggle
	
modifier with element:

	-effect-ripple_element
	-effect-ripple_super
